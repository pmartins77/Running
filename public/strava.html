// ✅ Afficher les activités sous forme de tableau function displayActivities(activities) { const activitiesList = document.getElementById("activitiesList"); activitiesList.innerHTML = ""; // Nettoyage avant ajout activities.forEach(activity => { const row = document.createElement("tr"); row.innerHTML = ` ${new Date(activity.date).toLocaleDateString()} ${activity.name || "N/A"} ${(activity.distance / 1000).toFixed(2)} ${(activity.elapsed_time / 60).toFixed(0)} ${(activity.average_speed * 3.6).toFixed(2)} `; activitiesList.appendChild(row); }); }
